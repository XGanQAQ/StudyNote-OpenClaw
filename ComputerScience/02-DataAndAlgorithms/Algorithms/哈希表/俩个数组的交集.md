```cs
    public int[] Intersection(int[] nums1, int[] nums2)
    {
        HashSet<int> set = new HashSet<int>();
        foreach (var num in nums1)
        {
            set.Add(num);
        }
        List<int> result = new List<int>();
        foreach (var num in nums2)
        {
            if (set.Contains(num))
            {
                result.Add(num);
                set.Remove(num); // 防止重复添加
            }
        }
        return result.ToArray();
    }
```
主要是哈希集合的应用
