
## 前序
迭代写法
```cs
    vector<int> preorderTraversal(TreeNode* root) {
        vector<int> res;
        stack<TreeNode*> sta;
        TreeNode* current = root;
        while (!sta.empty() || current != nullptr) {
            while (current != nullptr) {
                res.push_back(current->val);
                sta.push(current);
                current = current->left;
            }

            current = sta.top();
            sta.pop();
            current = current->right;
        }
        return res;
    }
```

## 中序

迭代写法
```cs
class Solution {
public:
    vector<int> inorderTraversal(TreeNode* root) {
        vector<int> res;
        stack<TreeNode*> st;
        TreeNode* curr = root;

// 当前节点不为空，代表还有节点可推入
// 栈不为空代表有节点可回溯
        while (curr != nullptr || !st.empty()) {
// 循环深入
            while (curr != nullptr) {
                st.push(curr);
                curr = curr->left;
            }

// 当搜索到深的时候
            curr = st.top();
            st.pop();
            res.push_back(curr->val);

            curr = curr->right;
        }

        return res;
    }
};
```
