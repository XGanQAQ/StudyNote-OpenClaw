### 曼哈顿距离定义：

二维平面中，点 `(x1, y1)` 和 `(x2, y2)` 的曼哈顿距离是：
```
|x1 - x2| + |y1 - y2|
```

## 代码循环标记曼哈顿距离

### 暴力遍历法
```python
# 示例 Python 代码
m = 5  # 行数
n = 5  # 列数
d = 2  # 曼哈顿距离阈值

# 初始化数组
grid = [[0 for _ in range(n)] for _ in range(m)]

# 选中位置
x0, y0 = 2, 2

# 标记符合条件的元素
for i in range(m):
    for j in range(n):
        if abs(i - x0) + abs(j - y0) < d:
            grid[i][j] = 1  # 用 1 进行标记

# 打印数组查看结果
for row in grid:
    print(row)

```

### 优化法
```cpp
#include <iostream>
#include <vector>
#include <cmath>
using namespace std;

int main() {
    int m = 5; // 行数
    int n = 5; // 列数
    int d = 2; // 曼哈顿距离阈值
    int x0 = 2, y0 = 2; // 选中元素的位置

    vector<vector<int>> grid(m, vector<int>(n, 0)); // 初始化数组

    for (int i = max(0, x0 - d + 1); i <= min(m - 1, x0 + d - 1); ++i) {
        int rowDist = abs(i - x0);
        int maxColDist = d - 1 - rowDist;
        
        int left = max(0, y0 - maxColDist);
        int right = min(n - 1, y0 + maxColDist);

        for (int j = left; j <= right; ++j) {
            grid[i][j] = 1; // 标记
        }
    }

    // 打印结果
    for (const auto& row : grid) {
        for (int val : row) {
            cout << val << " ";
        }
        cout << endl;
    }

    return 0;
}

```
